create materialized view view_materialized as (

select p.descricao "nome_do_projeto", count(atividades.id) "atividades_realizadas", 
f.nome "funcionario_respons√°vel"
from projetos p

join atividadesprojetos ap

join atividades

on p.funcionario_responsavel_id = f.id
